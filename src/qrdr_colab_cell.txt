def normalize_species(name):
    if not name:
        return None
    n = str(name).upper()
    if "COLI" in n:
        return "ECOLI"
    if "KLEB" in n or ("PNEUMO" in n and "STREP" not in n):
        return "KLEBSIELLA"
    if "SALMON" in n:
        return "SALMONELLA"
    if "PSEUDO" in n or "AERUG" in n:
        return "PSEUDOMONAS"
    if "STAPH" in n or "AUREUS" in n:
        return "STAPHYLOCOCCUS"
    return None


# ═══════════════════════════════════════════════════════════════════════════
# FASTA + FTPS HELPERS
# ═══════════════════════════════════════════════════════════════════════════
# Gen ismi pattern'leri — FASTA header'da bunlardan birini ara
GYRA_PATTERNS = ["[gyrA]", " gyrA ", "DNA gyrase subunit A"]
PARC_PATTERNS = ["[parC]", " parC ", "DNA topoisomerase IV subunit A",
                 "[grlA]", " grlA "]  # Stafilo için grlA = parC eşdeğeri


def stream_faa(gid: str, timeout=60):
    """
    Bir genome'un protein FASTA dosyasını streaming olarak indir.
    Birden çok dosya adı varyantını sırayla dener (BV-BRC dokümantasyonu
    tutarsız: bazı genomelarda .faa, bazılarında .PATRIC.faa var).
    Sadece gyrA + parC/grlA sekanslarını çıkar, gerisini at.
    """
    seqs = {"gyra": "", "parc": ""}
    # Dosya adı varyantları — sırayla dene
    candidate_paths = [
        f"/genomes/{gid}/{gid}.PATRIC.faa",
        f"/genomes/{gid}/{gid}.faa",
        f"/genomes/{gid}/{gid}.RefSeq.faa",
    ]

    buf = None
    try:
        ftps = FTP_TLS(FTP_HOST, timeout=timeout)
        ftps.login()
        ftps.prot_p()
        ftps.set_pasv(True)
        for path in candidate_paths:
            tmp = io.BytesIO()
            try:
                ftps.retrbinary(f"RETR {path}", tmp.write)
                # Başarılı + boş değil
                if tmp.tell() > 100:
                    buf = tmp
                    break
            except Exception:
                continue
        ftps.quit()
    except Exception:
        return seqs

    if buf is None:
        return seqs

    # FASTA parse
    try:
        text = buf.getvalue().decode("utf-8", errors="ignore")
    except Exception:
        return seqs
    if not text:
        return seqs

    current_gene = None
    current_seq  = []
    for line in text.split("\n"):
        if line.startswith(">"):
            # Yeni record başladı → eskisini sakla
            if current_gene and current_seq:
                seqs[current_gene] = "".join(current_seq)
            current_seq = []
            # Gen ismini parse et
            line_lower = line.lower()
            if any(p.lower() in line_lower for p in GYRA_PATTERNS):
                current_gene = "gyra"
            elif any(p.lower() in line_lower for p in PARC_PATTERNS):
                current_gene = "parc"
            else:
                current_gene = None
        elif current_gene:
            current_seq.append(line.strip())
    # Son record
    if current_gene and current_seq:
        seqs[current_gene] = "".join(current_seq)

    return seqs


# ═══════════════════════════════════════════════════════════════════════════
# MUTATION PARSE
# ═══════════════════════════════════════════════════════════════════════════
def find_anchor_offset(seq, anchor):
    if not seq or not anchor:
        return None
    pos = seq.find(anchor)
    if pos != -1:
        return pos
    for sub_len in [40, 30, 25, 20, 15]:
        pos = seq.find(anchor[:sub_len])
        if pos != -1:
            return pos
    return None


def parse_mutations(aa_seq, anchor, anchor_start_pos, mutations):
    out = {key: 0 for key in mutations}
    if not aa_seq or len(aa_seq) < 50:
        return out
    seq_pos_at_anchor = find_anchor_offset(aa_seq, anchor)
    if seq_pos_at_anchor is None:
        return out
    for key, (ref_pos, ref_aa, mut_aa) in mutations.items():
        seq_pos = seq_pos_at_anchor + (ref_pos - anchor_start_pos)
        if 0 <= seq_pos < len(aa_seq) and aa_seq[seq_pos] == mut_aa:
            out[key] = 1
    return out


def process_genome(gid: str, species_key: str):
    """Bir genome'u indir, parse et, mutation dict döndür."""
    cfg = QRDR_CATALOG.get(species_key)
    base = {"Genome ID": gid, "species_key": species_key}
    # Tüm mutation kolonları 0
    for c in ALL_MUT_COLS:
        base[c] = 0
    if not cfg:
        return base, False  # işlendi=False
    seqs = stream_faa(gid)
    found_any = bool(seqs.get("gyra") or seqs.get("parc"))
    if not found_any:
        return base, False
    # gyrA mutations
    base.update(parse_mutations(
        seqs.get("gyra", ""),
        cfg["gyrA_anchor"], cfg["gyrA_anchor_start"], cfg["gyrA_mutations"]
    ))
    # parC mutations
    base.update(parse_mutations(
        seqs.get("parc", ""),
        cfg["parC_anchor"], cfg["parC_anchor_start"], cfg["parC_mutations"]
    ))
    return base, True

# Checkpoint
done_ids = set()
if os.path.exists(OUT_FILE):
    try:
        old = pd.read_parquet(OUT_FILE)
        done_ids = set(old["Genome ID"].astype(str).tolist())
        # Bilgi: indirilen ama veri çekilemeyen kayıtları da say
        n_with_data = (old[ALL_MUT_COLS].sum(axis=1) > 0).sum()
        print(
            f"♻️  Checkpoint: {len(done_ids):,} kayıt zaten işlenmiş "
            f"({n_with_data:,} mutasyonlu)"
        )
    except Exception:
        old = None
else:
    old = None

remaining_df = proc_df[~proc_df["Genome ID"].isin(done_ids)]
print(f"📥 Yeni çekilecek: {len(remaining_df):,}")

if len(remaining_df) == 0:
    print("✅ Hepsi önbellekte.")
else:
    todo = remaining_df[["Genome ID", "species_key"]].values.tolist()
    new_rows = []
    counter  = {"done": 0, "with_data": 0, "fail": 0}
    lock     = threading.Lock()
    last_save = 0
    SAVE_EVRY = 1000
    t_start = time.time()

    def worker(args):
        gid, sp = args
        try:
            entry, ok = process_genome(gid, sp)
            return entry, ok
        except Exception:
            entry = {"Genome ID": gid, "species_key": sp}
            for c in ALL_MUT_COLS:
                entry[c] = 0
            return entry, False

    def flush(rows_so_far):
        """Önceki kayıtlarla birleştir ve parquet'e yaz."""
        merged = list(rows_so_far)
        if old is not None:
            merged = old.to_dict("records") + merged
        try:
            pd.DataFrame(merged).to_parquet(OUT_FILE, index=False)
        except Exception as e:
            # Drive'a yazılamadıysa /tmp'ye yaz
            tmp = "/tmp/v35_qrdr_mutations.parquet"
            pd.DataFrame(merged).to_parquet(tmp, index=False)
            print(f"\n   ⚠️  Drive yazma hatası, /tmp'ye kayıt: {tmp}")

    with ThreadPoolExecutor(max_workers=N_THREADS) as ex:
        futures = {ex.submit(worker, a): a for a in todo}
        for fut in as_completed(futures):
            entry, ok = fut.result()
            with lock:
                new_rows.append(entry)
                counter["done"] += 1
                if ok:
                    counter["with_data"] += 1
                else:
                    counter["fail"] += 1
                # Periyodik kayıt
                if counter["done"] - last_save >= SAVE_EVRY:
                    flush(new_rows)
                    last_save = counter["done"]
                # İlerleme
                if counter["done"] % 50 == 0:
                    elapsed = time.time() - t_start
                    rate    = counter["done"] / max(elapsed, 1)
                    eta     = (len(todo) - counter["done"]) / max(rate, 1)
                    pct     = counter["done"] / len(todo) * 100
                    print(
                        f"\r   %{pct:5.1f} "
                        f"({counter['done']:,}/{len(todo):,}) "
                        f"✓{counter['with_data']:,} "
                        f"✗{counter['fail']:,} | "
                        f"Hız: {rate:.1f}/s | ETA: {eta/3600:.1f}h",
                        end="", flush=True,
                    )

    # Final kayıt
    flush(new_rows)
    print()

# Yeniden oku ve özet ver

        n_pos = int(sub[c].sum())
        pct = n_pos / max(len(sub), 1) * 100

            bar = "█" * min(int(pct / 2), 40)
            print(f"      {c:18s}: {n_pos:5d} ({pct:5.2f}%) {bar}")
